"Tutorial" }}{{ $label = "Tutorials" }} {{- end -}} {{- $isActive := eq $currentCat $slug -}} {{- $termPage := site.GetPage (printf "/categories/%s" $slug) -}} {{- if $termPage }} {{ $label }} {{- else }} {{ $label }} {{- end -}} {{- end }}
